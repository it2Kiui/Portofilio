local Math = {}
Math.__index = Math
 
--[[ -- Raw
    {
        1, 2, 5,
        2, 9, 6,
        8, 4, 2,
    }
]]
 
--[[ -- Result
    {
        rows = {
            [1] = {1, 2, 5}
            [2] = {2, 9, 6}
            [3] = {8, 4, 2}
        }
        columns = {
            [1] = {1, 2, 8}
            [2] = {2, 9, 4}
            [3] = {5, 6, 2}
        }
    }
]]
 
type matrixType = typeof(Math) & {
    rawTable: {},
    matrixTable: {
        rows: {},
        columns: {},
    },
    rows: number,
    columns : number,
}
 
local function _genrateMatrix(rows : number, columns : number, rawTable)
    local matrix = {
        rows = {},
        columns = {},
    }
 
    for r = 1, rows do
        local row = {}
        for c = 1, columns do
            local index = (r - 1) * columns + c
            table.insert(row, rawTable[index])
        end
        matrix.rows[r] = row
    end
 
    for c = 1, columns do
        local column = {}
        for r = 1, rows do
            table.insert(column, matrix.rows[r][c])
        end
        matrix.columns[c] = column
    end
 
    return matrix
end
 
function Math.new(rows : number, columns : number, rawTable)
    local self : matrixType = setmetatable({}, Math)
    
    self.rows = rows
    self.columns = columns
    self.rawTable = rawTable
    
    local matrix = _genrateMatrix(rows, columns, rawTable)
    self.matrixTable = matrix
    
    return self
end
 
----------------------------------------------
---------------  PRIVATE  --------------------
----------------------------------------------
 
function Math._getRow(self: matrixType, index)
    return self.matrixTable.rows[index]
end
 
function Math._getColumn(self: matrixType, index)
    return self.matrixTable.columns[index]
end
 
function Math._toRawTable(self: matrixType)
    local result = {}
 
    for r = 1, self.rows do
        for c = 1, self.columns do
            table.insert(result, self.matrixTable.rows[r][c])
        end
    end
 
    return result
end
 
function Math._inverse(self: matrixType) :: matrixType
    local raw = self:_toRawTable()
    local inverted = {}
 
    for i = 1, #raw do
        inverted[i] = -raw[i]
    end
 
    return Math.new(self.rows, self.columns, inverted)
end
 
----------------------------------------------
---------------  PUBLIC  ---------------------
----------------------------------------------
 
function Math.addMatrices(self: matrixType, matrixObject: matrixType) :: matrixType
    assert(
        self.rows == matrixObject.rows and self.columns == matrixObject.columns,
        "Matrices must have the same dimensions"
    )
    
    local result = {}
    
    local rawTable1 = self:_toRawTable()
    local rawTable2 = matrixObject:_toRawTable()
    
    for i = 1, #rawTable1 do
        result[i] = rawTable1[i] + rawTable2[i]
    end
    
    return Math.new(self.rows, self.columns, result)
end
 
function Math.subtractMatrices(self: matrixType, matrixObject: matrixType)
    return self:addMatrices(matrixObject:_inverse())
end
 
function Math.multiplyMatrices(self: matrixType, matrixObject: matrixType) :: matrixType
    assert(self.columns == matrixObject.rows, "Invalid dimensions for multiplication")
 
    local result = {}
 
    for r = 1, self.rows do
        for c = 1, matrixObject.columns do
            local sum = 0
 
            for k = 1, self.columns do
                sum += self.matrixTable.rows[r][k] * matrixObject.matrixTable.rows[k][c]
            end
 
            table.insert(result, sum)
        end
    end
 
    return Math.new(self.rows, matrixObject.columns, result)
end
 
function Math.transpose(self: matrixType) :: matrixType
    local result = {}
 
    for c = 1, self.columns do
        for r = 1, self.rows do
            table.insert(result, self.matrixTable.rows[r][c])
        end
    end
 
    return Math.new(self.columns, self.rows, result)
end
 
-- Made to print the matrix
function Math.drawMatrix(self: matrixType)
    for r = 1, self.rows do
        local rowStr = ""
 
        for c = 1, self.columns do
            rowStr ..= tostring(self.matrixTable.rows[r][c]) .. "\t"
        end
 
        print("{    " .. rowStr .. "}")
    end
end
 
return Math
