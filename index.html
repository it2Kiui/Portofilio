Remotes.luau 
--!strict

local Players = game:GetService("Players")

local Remotes = {}

type PackedArgs = {
	n: number,
	[number]: any
}

type PlayerCalls = {
	[Player]: { number }
}

type Callback = (Player, ...any) -> ()
type Disconnect = () -> ()

local traffic = setmetatable({} :: { [RemoteEvent]: PlayerCalls }, {
	__mode = "k"
})

local function CheckArgs(args: PackedArgs, expected: { string }): boolean
	if args.n ~= #expected then
		return false
	end

	for i = 1, args.n do
		local value = args[i]
		local wanted = expected[i]
		local got = typeof(value)

		if got == wanted then
			continue
		end

		if got == "Instance" and value:IsA(wanted) then
			continue
		end

		return false
	end

	return true
end

Players.PlayerRemoving:Connect(function(player: Player)
	for _, calls in traffic do
		calls[player] = nil
	end
end)

function Remotes.Wrap(remote: RemoteEvent,
	expected: { string },
	callback: Callback,
	maxCalls: number?,
	window: number?,
	onLimitExceeded: ((Player) -> ())?
): Disconnect
	assert(not traffic[remote], "remote already wrapped: " .. remote.Name)

	local limit = maxCalls or 5
	local span = window or 1
	assert(limit > 0, "maxCalls must be higher than 0")
	assert(limit % 1 == 0, "maxCalls must be an integer")
	assert(span > 0, "window must be higher than 0")
	
	local calls: PlayerCalls = {}
	traffic[remote] = calls

	local connection = remote.OnServerEvent:Connect(function(player: Player, ...: any)
		if player.Parent ~= Players then
			return
		end

		local now = os.clock()
		local list = calls[player]

		if not list then
			list = {}
			calls[player] = list
		end

		local oldCount = #list
		local write = 1

		for i = 1, oldCount do
			local sentAt = list[i]

			if now - sentAt < span then
				list[write] = sentAt
				write += 1
			end
		end

		for i = write, oldCount do
			list[i] = nil
		end

		local count = write - 1

		if count >= limit then
			if onLimitExceeded then
				onLimitExceeded(player)
			else
				warn(player.Name .. " rate limited on " .. remote.Name)
			end

			return
		end

		count += 1
		list[count] = now

		local args = table.pack(...)

		if not CheckArgs(args, expected) then
			return
		end

		callback(player, table.unpack(args, 1, args.n))
	end)

	return function()
		if connection.Connected then
			connection:Disconnect()
		end

		traffic[remote] = nil
		table.clear(calls)
	end
end

return Remotes

RemoteWrapper: Rate-Limited Remote Sanitizer

A wrapper for every RemoteEvent in production. Handles type validation and rate limiting in one place; no repeated boilerplate across systems.

  • Remotes.Wrap rejects calls that dont match the expected argument signature and blocks players who fire too fast. The rate limit window and call cap are configurable per-remote, with an optional onLimitExceeded callback for custom handling like kicks or telemetry.
  • Uses a sliding window for rate limiting; stale timestamps are swept on each fire, so counts always reflect the actual current window (no timers, no scheduled cleanup). PlayerRemoving handles memory automatically, and the weak keyed traffic table ensures that destroyed remotes don't leak.
  • The returned disconnect function is fully clean; listener disconnected, call table cleared, remote removed from the registry.
